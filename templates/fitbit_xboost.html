Creating Function for RMS¶

In [2]:
def rmse(y1, y2):
    ss = 0
    for x1, x2 in zip(y1, y2):
        x1 = float(x1)
        x2 = float(x2)
        ss += pow((x1 - x2), 2)
        
    m = ss / len(y1)
    return math.sqrt(m)
